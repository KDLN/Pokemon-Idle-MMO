      control: 'select',
      options: ['primary', 'secondary', 'ghost', 'danger', 'pokeball'],

    size: {
      control: 'select',
      options: ['sm', 'md', 'lg', 'icon'],
      description: 'Size of the button',

      control: 'boolean',
      description: 'Shows loading spinner and disables interaction',

      control: 'text',
